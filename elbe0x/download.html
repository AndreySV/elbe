Debian 6.0 (squeeze)

Installation using apt package management

# su -
# cd /tmp
# wget http://debian.linutronix.de/elbe/elbe-repo.pub
# apt-key add elbe-repo.pub
# echo "deb http://debian.linutronix.de/elbe squeeze main" >> /etc/apt/sources.list
# apt-get update
# apt-get install elbe
